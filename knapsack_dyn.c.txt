#include<stdio.h>
#include<stdlib.h>


void swap(float *x, float*y)
{
    float temp;
    temp=*x;
    *x=*y;
    *y=temp;
}
void knapsack(float x[],float p[],float w[],int n,float m)
{
    int i,j;
    for(i=1;i<=n;i++)
        x[i]=0.0;
    for(i=1;i<n;i++)
        for(j=1;j<n;j++)
            if((p[i]/w[i])<(p[j+1]/w[j+1])) 
            {
                swap(&p[i],&p[j+1]);
                swap(&w[i],&w[j+1]);
            }   
    for(i=1;i<=n;i++)
    {
        if(w[i]>m)
            return;
        x[i]=1;
        m=m-w[i];    
    }        
    return;
}
void main()
{
    float x[20],p[20],w[20],profit=0.0,m;
    int n,i;
    printf("\nEnter no. of items:");
    scanf("%d",&n);
    printf("\nEnter the capacity of stack:");
    scanf("%f",&m);
    printf("\nEnter profits and weights of each item:\n");
    printf("\nProfit: Weight:\n");
    for(i=1;i<=n;i++)
        scanf("%f %f",&p[i],&w[i]);
    knapsack(x,p,w,n,m);
    for(i=1;i<=n;i++)
        profit+=p[i]*x[i];
    printf("\nTotal Profit:%f",profit);
    return;        
}
